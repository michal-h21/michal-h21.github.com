    Node pNode = dom.getFirstChild();
    if( pNode.hasChildNodes() ){
       boolean drop = true;
       Node child = pNode.getFirstChild();
       while( child != null ){
          short type = child.getNodeType();

              (type == Node.ELEMENT_NODE)
            ||
              (type == Node.CDATA_SECTION_NODE)
            ||
              (type == Node.TEXT_NODE)
            &&

          ){
