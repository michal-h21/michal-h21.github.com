4.5 Undo Empty Scripts

<..undo empty scripts..>
 <script element="span::msup" >
    <set name="m-sub-sup" >
      <.open xslt script.>
      <.undo if empty su.>
      <.close xslt script.>
    </set>
    <xslt name="." xml="." xsl="m-sub-sup" />
    <.superscript 2 and 3 into verbose.>
 </script>
 <script element="span::msub" >
   <.sub sup script.>
 </script>
 <script element="span::msubsup" >
   <.subsup prime.>
   <.sub sup script.>
   <.sub superscript 2 and 3 into verbose.>
 </script>
