<..latex math..>
 \Configure{equation}
    {}    {\HCode{<title>}}   {\HCode{</title>}}
 \ConfigureEnv{equation}
    {\IgnorePar\EndP
     \HCode{<div rend="equation"><p rend="equation">}}
    {\HCode{</p></div>}\ShowPar\par}
