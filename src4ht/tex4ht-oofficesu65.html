 \ifx \find:set:mgroup\:UnDef
    \def\find:set:mgroup#1#2#3{\bgroup
       \let\HPage\:gobble
       \let\EndHPage\:gobble
       \def\use@mathgroup##1##2##3{\relax
         \tmp:cnt=##2%
         \xdef\:temp####1{\noexpand\Configure{MathClass}{#2}%
             {}{}{}{\mathchar"0\the\tmp:cnt ####1}}%
       }%
       \setbox\tmp:bx=\hbox{$#1$}\setbox\tmp:bx=\hbox{}%
       \set:mgroup#3{}{}%
     \egroup
     }
 \fi
