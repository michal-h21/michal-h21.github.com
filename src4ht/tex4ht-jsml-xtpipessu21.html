     (@class = ’inline-math’)

     (count( <.math content element.> ) = 1)
   and
     child::mrow[ @class=’MathClass-open-close’ ]
     / child::mrow[
        not( child::*[
          not(self::mo) and not(self::mn-group-s)
          and not(self::mi) and not(self::BREAK)
          and not( @title=’speech-extra’ )
        ])
     ]
 ]" >

     <.shallow delimited expression.>
   </xsl:copy>
 </xsl:template>
