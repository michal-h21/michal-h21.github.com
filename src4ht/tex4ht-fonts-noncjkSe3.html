 class Pattern{
      String [] p;
    Pattern( String p ){
      p = p.trim().replaceAll("\\s+"," ");
      this.p = p.split(" ");
    }
    boolean isWithin( String s ){
      int from = 0;
      for(int i=0; i<p.length; i++){
         int idx = s.indexOf(p[i], from);
         if( idx == -1 ){ return false; }
         from = i;
      }
      return true;
    }
