2.4 Color

<..configure mathplayer color..>
 \let\a:txt:textcolor\a:textcolor
 \let\b:txt:textcolor\b:textcolor
 \Configure{textcolor}
    {\ifmathml
       \Configure{color}%
          {\Configure{SetHColor}
               {\HCode{<m:mstyle color="\HColor">}}}%
     \else \a:txt:textcolor \fi
    }
    {\ifmathml \HCode{</m:mstyle>}\else \b:txt:textcolor \fi}
