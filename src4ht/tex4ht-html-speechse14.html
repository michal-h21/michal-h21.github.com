   \IgnorePar\EndP  \HAssign\Verb:N=0

          <pre class="#1"\a:LRdir>}\EndNoFonts}
 \def\endenv:verb{\NoFonts

 \def\start:fenv#1{\:xhtml{\IgnorePar
