 \Configure{eqnarray}
    {\HCode{\string\begin{eqnarray\ifx
        \@seqncr\@eqncr\else *\fi}}}
    {\HCode{\string\end{eqnarray\ifx
        \@seqncr\@eqncr\else *\fi}}}
    {}
    {\HCode{\string\\\Hnewline}}
    {\ifnum \HCol>1
       \ifnum \HCol=4
         \HCode{\%}%
         <.start mtext space.>\PauseMathClass
         \mtexttrue
       \fi \HCode{\a:math:amp}%
     \fi}
    {\ifnum \HCol=4 \mtextfalse
       \EndPauseMathClass <.end mtext space.>\fi }
