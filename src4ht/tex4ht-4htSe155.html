\documentclass{article} 
   \usepackage[sort&compress]{natbib} 
\begin{document}

double citations are not \cite{bar,foo}. 
 
\begin{thebibliography}{99} 
\bibitem[1]{foo} foofoo 
\bibitem[2]{bar} barbar 
\end{thebibliography} 
\end{document}
