 \def\list:save{%
    \let\SVempty:D:T:D=\empty:D:T:D
    \let\SVend:DL=\end:DL
    \let\SV:DT=\:DT
    \let\SV:DD=\:DD }
 \def\list:recall{%
    \global\let\empty:D:T:D=\SVempty:D:T:D
    \global\let\end:DL=\SVend:DL
    \global\let\:DT=\SV:DT
    \global\let\:DD=\SV:DD }

The \@newlistfalse inserted for handling the following case:

    \documentclass{article} 
    \begin{document} 
    \catcode‘\@=11 
    \begin{list}{}         %<-- missing argument 
    \item 
    \hshow{-----item:NEW------------------------->\if@newlist NEW\else OLD\fi}% 
    a 
    \hshow{-----/item:OLD------------------------>\if@newlist NEW\else OLD\fi}% 
    \end{list}
