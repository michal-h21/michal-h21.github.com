6.5 Eliminate Begin Script Marks

<..eliminate begin script marks..>
 <xsl:template match="span[
      (@class = ’begin-script’)
    and
      following-sibling::* [1] / child::*[
        (position() = 1)
        and
        (@class = ’mrow-base’)
        and
        ( normalize-space(.) = ’’ )
      ]
 ]"  >
 </xsl:template>
