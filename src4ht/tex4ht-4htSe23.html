8.2 Choose

<..plain,latex math..>+
 \def\atop:choose#1#2#3{\a:choose}
 \def\:temp{%
    \expandafter \ifx\csname a:choose\endcsname\relax \else
    \ifx\a:choose\empty \else
       \Configure{atopwithdelims}{\atop:choose}{}%
    \fi\fi
    \o:choose:}
 \HLet\choose=\:temp
 \NewConfigure{choose}{1}
 -_-_-
