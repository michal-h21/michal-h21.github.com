6.3 Word Fractions

<..constant fracs..>
 <xsl:when test="
    (string-length(
       normalize-space(child::span[ @class = ’mrow-numerator’ ][1])
      ) = 1)
    and
    (string-length(
      normalize-space(child::span[ @class = ’mrow-enumerator’ ][1])
      ) = 1)
 ">
    <.a := numerator; b := enumerator.>
    <xsl:choose>
       <xsl:when test="
            (translate($a,’123456789’,’’) != ’’)
            or
            (translate($b,’123456789’,’’) != ’’)
        " >
          <xsl:apply-templates select="*|@*|text()|comment()" />
       </xsl:when>
       <xsl:when test=" $a &lt; $b ">
          <xsl:attribute  name="class">
              <xsl:text>word-frac</xsl:text>
          </xsl:attribute>
          <.word numerator.>
          <.word enumerator.>
       </xsl:when>
       <xsl:otherwise>
          <xsl:apply-templates select="*|@*|text()|comment()" />
       </xsl:otherwise>
    </xsl:choose>
 </xsl:when>

<..a := numerator; b := enumerator..>
 <xsl:variable name="a">
    <xsl:value-of select="
       normalize-space(child::span[ @class = ’mrow-numerator’ ][1])
    " />
 </xsl:variable>
 <xsl:variable name="b">
    <xsl:value-of select="
       normalize-space(child::span[ @class = ’mrow-enumerator’ ][1])
    " />
 </xsl:variable>
