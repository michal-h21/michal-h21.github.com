 \let\no@subsection=\subsection
 \Def:Section\subsection{\ifnum \c:secnumdepth>\c@secnumdepth   \else

 \def\subsection{\rdef:sec{subsection}}

 \let\:likesubsection=\likesubsection
 \let\likesubsection=\:UnDef
