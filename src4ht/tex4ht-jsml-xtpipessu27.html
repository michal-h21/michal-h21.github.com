6.6 Undo Empty Scripts

<..special sub and super scripts..>
 <script element="msup" >
    <set name="m-sub-sup" >
      <.open xslt script.>
      <.undo if empty su.>
      <.close xslt script.>
    </set>
    <xslt name="." xml="." xsl="m-sub-sup" />
    <.superscript 2 and 3 into verbose.>
 </script>
 <script element="msub" >
   <.sub script.>
 </script>
 <script element="msubsup" >
   <.subsup prime.>
   <.sub sup script.>
   <.sub superscript 2 and 3 into verbose.>
 </script>
