   <set name="apostro" >
      <.open xslt script.>
      <.right apostrophe in prose.>
      <.delimiters in marks of enumerated lists.>
      <.remove empty paragraphs.>
      <.close xslt script.>
   </set>
   <xslt name="." xml="." xsl="apostro" />
 </script>
