<..remove scrip indicators from primes..>
 <xsl:template match="span[
     (@class = ’begin-script’)
   and
     following-sibling::*[ not(@title = ’speech-extra’) ] [1]
     / descendant-or-self::*[ not( self::PROS ) ][1]
     /
          child::span / child::span[
             (@class = ’ch 2032’) or (@class = ’ch 2033’)
                                  or (@class = ’ch 2034’)
     ]
 ]" >
   <BREAK SIZE="small"/>
 </xsl:template>
