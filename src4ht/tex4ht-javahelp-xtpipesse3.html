3 Long Tables

<..longtable script..>
 <script element="table::longtable" >
    <set name="longtbl" >
       <.open xslt script.>
       <.longtable templates.>
       <.close xslt script.>
    </set>
    <xslt name="." xml="." xsl="longtbl" />
 </script>
-_-_-

<..longtable templates..>
 <xsl:template match=" table[ @class=’longtable’ ]
                     / tr[ (normalize-space(.)=’’) ] " >
    <xsl:if test=" normalize-space(following-sibling::*) != ’’ ">
       <xsl:copy>
          <xsl:apply-templates select="*|@*|text()|comment()" />
       </xsl:copy>
    </xsl:if>
 </xsl:template>
